    constructor(comparator = (a, b) => a - b, data = []) {
        this.data = data;
        this.comparator = comparator;//æ¯”è¾ƒå™¨
        this.heapify();//å †åŒ–
    }

    heapify() {
        if (this.size() < 2) return;
        for (let i = Math.floor(this.size()/2)-1; i >= 0; i--) {
            this.bubbleDown(i);//bubbleDownæ“ä½œ
        }
    }

    peek() {
        if (this.size() === 0) return null;
        return this.data[0];//æŸ¥çœ‹å †é¡¶
    }

    offer(value) {
        this.data.push(value);//åŠ å…¥æ•°ç»„

    }

    poll() {

            return null;

        const result = this.data[0];

            this.data[0] = last;//äº¤æ¢ç¬¬ä¸€ä¸ªå…ƒç´ å’Œæœ€åä¸€ä¸ªå…ƒç´ 
            this.bubbleDown(0);//bubbleDownæ“ä½œ

    }

    bubbleUp(index) {
        while (index > 0) {
            const parentIndex = (index - 1) >> 1;//çˆ¶èŠ‚ç‚¹çš„ä½ç½®
            //å¦‚æœå½“å‰å…ƒç´ æ¯”çˆ¶èŠ‚ç‚¹çš„å…ƒç´ å°ï¼Œå°±äº¤æ¢å½“å‰èŠ‚ç‚¹å’Œçˆ¶èŠ‚ç‚¹çš„ä½ç½®
            if (this.comparator(this.data[index], this.data[parentIndex]) < 0) {

                index = parentIndex;//ä¸æ–­å‘ä¸Šå–çˆ¶èŠ‚ç‚¹è¿›è¡Œæ¯”è¾ƒ

                break;//å¦‚æœå½“å‰å…ƒç´ æ¯”çˆ¶èŠ‚ç‚¹çš„å…ƒç´ å¤§ï¼Œä¸éœ€è¦å¤„ç†
            }
        }
    }

    bubbleDown(index) {
        const lastIndex = this.size() - 1;//æœ€åä¸€ä¸ªèŠ‚ç‚¹çš„ä½ç½®
        while (true) {
            const leftIndex = index * 2 + 1;//å·¦èŠ‚ç‚¹çš„ä½ç½®

            if (

                this.comparator(this.data[leftIndex], this.data[findIndex]) < 0

                findIndex = leftIndex;
            }
            if (
                rightIndex <= lastIndex &&
                this.comparator(this.data[rightIndex], this.data[findIndex]) < 0
            ) {
                findIndex = rightIndex;
            }
            if (index !== findIndex) {
                this.swap(index, findIndex);//äº¤æ¢å½“å‰å…ƒç´ å’Œå·¦å³èŠ‚ç‚¹ä¸­valueå°çš„
                index = findIndex;

            }

    }

        [this.data[index1], this.data[index2]] = [this.data[index2], this.data[index1]];
    }

    size() {

    }
